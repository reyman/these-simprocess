\newcommand{\mymakelabel}[1]{%
    \begingroup
    \def\@tempa{#1}%
    \def\@tempb{\@itemlabel}%
    \ifx\@tempa\@tempb
    \endgroup
        \hss \llap{\textcolor{blue}{\rule{1em}{1em}}}%
    \else
    \endgroup
        \hss \llap{%
    \circledTask{#1}%
        }%
    \fi
}
